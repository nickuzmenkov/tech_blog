    Mangling: Hiding Methods and Attributes

    Nuclear option: making attribute invisible from outside class by starting it with double underscore (__):

    class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        # Starting attribute name with a '__' makes it only visible 
                # from within the class
                self.__id_number = id_number
        self.full_name = full_name
        self.department = department

    def summary(self) -> str:
                # Methods of the class can access the hidden attribute
        return f"{self.full_name} (ID {self.__id_number})"

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    print(john_doe.summary())
        # But attempt to access the hidden attribute from outside class 
        # raises AttributeError
    print(john_doe.__id_number)

    A more gentle (and preferred) option: marking property as hidden

    class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        # Starting attribute name with a '_' notes other users
                # that it must not be accessed and modified 
                self._id_number = id_number
        self.full_name = full_name
        self.department = department

    def summary(self) -> str:
        return f"{self.full_name} (ID {self._id_number})"

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    print(john_doe.summary())
    print(john_doe._id_number)
    💡 The same applies to methods.

    Chaining Method Calls

    If you need your class to have Pandas- or NumPy-like API supporting chaining method calls like

    import numpy as np

x = np.array([1, 2, 3, 4, 5, 6])
x = x.reshape((2, 3)).astype(float).mean()

    just return self in methods you want to be called in chain

    class Employee:
    def __init__(self, id_number: int, full_name: str, department: str, grade: int) -> None:
        self._id_number = id_number
        self.full_name = full_name
        self.department = department
        self.grade = grade

    def change_grade(self, new_grade):
        self.grade = new_grade
                # Return the instance back to allow calling methods
        # directly on the result
        return self  

    def change_department(self, new_department):
        self.department = new_department
                # Same applies here
        return self

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
        grade=3
    )
        # Call methods in chain
    john_doe.change_grade(4).change_department("IT")
        # Make sure changes are applied
    print(john_doe.grade)
    print(john_doe.department)

    Instance, Class, and Static Methods

    Class methods have access to instance methods and class attributes, and usually provide alternative constructors:

    class Employee:
    default_grade = 1

    def __init__(self, id_number: int, full_name: str, department: str, grade: int) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department
        self.grade = grade

    @classmethod
    def from_dict(cls, dict_):
                # Allow creating the object from a dictionary
        return cls(
            id_number=dict_["id_number"],
            full_name=dict_["full_name"],
            department=dict_["department"],
            grade=dict_.get("grade", default=cls.default_grade),
        )

if __name__ == "__main__":
    john_doe_dict = {
        "id_": 42,
        "full_name": "John Doe",
        "department": "HR",
    }
        # (!) No parentheses after the class name
    john_doe = Employee.from_dict(john_doe_dict)

    Instance methods (the regular ones, without any decorators) have access to instance attributes and all methods

    COMPANY_NAME = "MIDAS"

class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department

    def get_email(self) -> str:
        normalized_name = self.full_name.replace(" ", ".")
        return f"{normalized_name}@{self.department}.{COMPANY_NAME}.ai".lower()

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    john_doe_email = john_doe.get_email()
    print(john_doe_email)

    Static methods do not have access to any class fields

    COMPANY_NAME = "MIDAS"

class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department

    @staticmethod
    def get_email(full_name: str, department: str) -> str:
                # Static methods do not accept the 'self' parameter
        normalized_name = full_name.replace(" ", ".")
        return f"{normalized_name}@{department}.{COMPANY_NAME}.ai".lower()

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    john_doe_email = john_doe.get_email(
                full_name=john_doe.full_name,
                department=john_doe.department,
        )
    print(john_doe_email)

    Properties, Getters, and Setters

    Property is a method in attribute disguise

    COMPANY_NAME = "MIDAS"

class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department

    @property
    def email(self) -> str:
        normalized_name = self.full_name.replace(" ", ".")
        return f"{normalized_name}@{self.department}.{COMPANY_NAME}.ai".lower()

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
        # Despite 'email' is a method, we can access it like an attribute
        # (!) Do not use parentheses with properties
    print(john_doe.email)
    💡 It’s generally a bad idea to do heavy operations in property code as it will be executed each time the property is accessed.

    Property can have a setter. A setter is used to disallow invalid values, thus protecting the property.

    COMPANY_NAME = "MIDAS"
DEPARTMENTS = ["HR", "Sales", "IT"]

class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self._department = department

    @property
    def department(self) -> str:
        return self._department

    @department.setter
    def department(self, value: str):
        if value not in DEPARTMENTS:
            raise ValueError(f"Department must be one of {DEPARTMENTS}, but found: {value}.")
        self._department = value

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    john_doe.department = "Sales"
        # This will raise ValueError because the property is protected by the setter
    john_doe.department = "Finance"

    Magic Methods

    There are a whole bunch of magic methods. Those are mostly enable common operations disabled for custom classes by default, like comparison (><=), type conversion, indexing, calling, etc. 

    class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department

    def __repr__(self) -> str:
                # The '__repr__' method will override the clunky default string
                # displayed when we print the class instance
        return f"{self.full_name} ({self.department})"

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
        # This will now print a pretty string instead of default one
    print(john_doe)

    Let’s consider another example and enable comparison operations for our Employee class

    class Employee:
    def __init__(self, id_number: int, full_name: str, department: str, grade: int) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department
        self.grade = grade

    def __gt__(self, other) -> bool:
                # Short for 'greater than' called when two instances 
                # are compared with '>' or '<'
                # You can define the '__lt__' (less than) method instead
        return self.grade > other.grade

    def __ge__(self, other) -> bool:
                # Short for 'greater than or equals' called when two instances 
                # are compared with '>=' or '<='
                # You can define the '__le__' (less than or equals) method instead
        return self.grade >= other.grade

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
        grade=3,
    )
    jane_doe = Employee(
        id_number=43,
        full_name="Jane Doe",
        department="Sales",
        grade=5,
    )
        # You can now directly compare two instances of the 'Employee' class
    print(john_doe <= jane_doe)

    You can also enable the calling behaviour (calling means using parentheses directly on the instance of the class):

    class Processor:
    def __init__(self, operations) -> None:
        self.operations = operations

    @staticmethod
    def apply_operation(data, operation):
        return operation(data)

    def __call__(self, data):
                # This method is called when instance of the class is called
        for operation in self.operations:
            data = self.apply_operation(data, operation)
        return data

if __name__ == "__main__":
    processor = Processor(
        operations=[
            lambda x: x + 10,
            lambda x: x * 2
        ]
    )
        # You can now call the processor directly
    print(processor(42))
    print(processor(12))

    Inheritance

    The simplest example is a class that inherits everything from its parent class, but also adds its own, unique methods or attributes

    import textwrap

COMPANY_NAME = "MIDAS"

class BaseEmployee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department
        self.email = self.get_email()

    def get_email(self) -> str:
        normalized_name = self.full_name.replace(" ", ".")
        return f"{normalized_name}@{self.department}.{COMPANY_NAME}.ai".lower()

class HREmployee(BaseEmployee):
        # This class inherits the constructor and 'get_email' methods
        # and also adds a new 'compose_vacancy' method

    def compose_vacancy(self, position: str, description: str) -> str:
                # Note that we can reference parent methods and attributes
                # From a child class
        vacancy_text = f"""
        Company {COMPANY_NAME} is searching a new {position}.

        Description
        -----------
        {description}

        Contacts
        --------
        {self.full_name} ({self.email})
        """
        return textwrap.dedent(vacancy_text)

if __name__ == "__main__":
    john_doe = HREmployee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    print(john_doe.compose_vacancy(position="ML Engineer", description="<Description>"))

    You can override parent method completely by simply using the same method name. If you need to reference parent’s behaviour from this method, you can do so by calling the super() object which returns parent class

    Abstract Classes and Interfaces

    Abstract classes are classes that define some behaviour but also require all child classes to implement some mandatory behaviour (i.e. methods or properties). If this behaviour is not implemented, the child class is invalid and cannot be initialised

    from abc import ABC, abstractmethod

COMPANY_NAME = "MIDAS"

class BaseEmployee(ABC):
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department
        self.email = self.get_email()

    @abstractmethod
    def get_email(self) -> str:
        pass

class HREmployee(BaseEmployee):
    def __init__(self, id_number: int, full_name: str) -> None:
        super().__init__(id_number=id_number, full_name=full_name, department="HR")

    def get_email(self) -> str:
        normalized_name = self.full_name.replace(" ", ".")
        return f"{normalized_name}@{COMPANY_NAME}.ai".lower()

if __name__ == "__main__":
    john_doe = HREmployee(
        id_number=42,
        full_name="John Doe",
    )

    Properties can also be abstract

    from abc import ABC, abstractmethod

COMPANY_NAME = "MIDAS"

class BaseEmployee(ABC):

        self.full_name = full_name
        self.department = department

    @property
    @abstractmethod
    def email(self) -> str:
        pass

        super().__init__(id_number=id_number, full_name=full_name, department="HR")

    @property
    def email(self) -> str:
        normalized_name = self.full_name.replace(" ", ".")
        return f"{normalized_name}@{COMPANY_NAME}.ai".lower()

if __name__ == "__main__":
    john_doe = HREmployee(
        id_number=42,

    )
    print(john_doe.email)

    You can also inherit from abstract class to add some extra behaviour but leave the class abstract too. Think of it as a more specific abstraction, that’s still an abstraction

    import textwrap
from abc import ABC, abstractmethod

COMPANY_NAME = "MIDAS"

class BaseEmployee(ABC):
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self.id_number = id_number
        self.full_name = full_name
        self.department = department

    @property
    @abstractmethod
    def email(self) -> str:
        pass

class BaseHREmployee(BaseEmployee, ABC):
    def __init__(self, id_number: int, full_name: str) -> None:
        super().__init__(id_number=id_number, full_name=full_name, department="HR")

    @abstractmethod
    def compose_vacancy(self, position: str, description: str) -> str:
        pass

class HREmployee(BaseHREmployee):
    def __init__(self, id_number: int, full_name: str) -> None:
        super().__init__(id_number=id_number, full_name=full_name)

    @property
    def email(self) -> str:
        normalized_name = self.full_name.replace(" ", ".")
        return f"{normalized_name}@{COMPANY_NAME}.ai".lower()

    def compose_vacancy(self, position: str, description: str) -> str:
        vacancy_text = f"""
        Company {COMPANY_NAME} is searching a new {position}.

        Description
        -----------
        {description}

        Contacts
        --------
        {self.full_name} ({self.email})
        """
        return textwrap.dedent(vacancy_text)

if __name__ == "__main__":
    john_doe = HREmployee(
        id_number=42,
        full_name="John Doe",
    )
    print(john_doe.compose_vacancy(position="ML Engineer", description="<Description>"))

    Interfaces are special types of abstract classes that don’t define any default behaviour whatsoever. Interfaces are useful blueprints that show what properties and methods each child class has

    from abc import ABC, abstractmethod
from typing import List, Dict

class BaseProcessor(ABC):
    @property
    @abstractmethod
    def input_values(self) -> List[str]:
        pass

    @property
    @abstractmethod
    def output_values(self) -> List[str]:
        pass

    @abstractmethod
    def __call__(self, data: Dict[str, float]) -> Dict[str, float]:
        pass

class MeanProcessor(BaseProcessor):
    def __init__(self, input_values: List[str], output_value: str = "mean") -> None:
        self._input_values = input_values
        self._output_value = output_value

    @property
    def input_values(self) -> List[str]:
        return self._input_values

    @property
    def output_values(self) -> List[str]:
        return [self._output_value]

    def __call__(self, data: Dict[str, float]) -> Dict[str, float]:
        result = sum(data[x] for x in self._input_values) / len(self._input_values)
        return {self._output_value: result, **data}

if __name__ == "__main__":
    processors = [
        MeanProcessor(input_values=["a", "b"], output_value="mean_a_b"),
        MeanProcessor(input_values=["a", "c"], output_value="mean_a_c"),
    ]
    data = {"a": 10, "b": 20, "c": 30}

    for processor in processors:
        data = processor(data)

    print(data)

    Special Classes: Dataclasses and Enums

    Enums allow only certain range of options

    from enum import Enum

class DepartmentKind(Enum):
    SALES = "SALES"
    HR = "HR"
    IT = "IT"

class Employee:
    def __init__(self, id_number: int, full_name: str, department: str) -> None:
        self._id_number = id_number
        self.full_name = full_name
                # If department is not one of 'SALES', 'HR', or 'IT'
                # enum will raise a human-readable exception for us
        self._department = DepartmentKind(department)

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department="HR",
    )
    jane_doe = Employee(
        id_number=42,
        full_name="Jane Doe",
        department="Marketing",
    )

    Dataclasses are useful when the class has little to no special behaviour and just stores data in attributes

    from dataclasses import dataclass
from enum import Enum

class DepartmentKind(Enum):
    SALES = "SALES"
    HR = "HR"
    IT = "IT"

@dataclass
class Employee:
        # Here, employee doesn't have any behaviour but rather only stores data
    id_number: int
    full_name: str
    department: DepartmentKind

if __name__ == "__main__":
    john_doe = Employee(
        id_number=42,
        full_name="John Doe",
        department=DepartmentKind.HR,
    )
    💡 Note that type annotations are mandatory for dataclasses.
